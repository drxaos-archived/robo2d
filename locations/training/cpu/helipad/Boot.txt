import com.robotech.military.api.*;

public class Boot implements Program {

    @Override
    public void run(Host host) {
        Console console = host.getDevice(Console.class);
        Door door = host.getDevice(Door.class);

        door.select("helipadDoor"); // door should be selected by secret key

        console.println("Initialization...");
        console.println("Available commands: (\"open\", \"close\")");

        while (true) {
            String in = console.read("cmd> ");

            if (in.equals("open")) { // Open selected door
                door.open();

            } else if (in.equals("close")) { // Close selected door
                door.close();

            } else {
                console.println("Unknown command: " + in);
            }
        }
    }
}
