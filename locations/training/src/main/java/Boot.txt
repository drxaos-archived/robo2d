import com.robotech.military.api.*;

public class Boot implements Program {

    Robot robot;
    Chassis chassis;
    Radar radar;
    Gps gps;

    @Override
    public void run(Robot robot) {
        this.robot = robot;
        chassis = robot.getDevice(Chassis.class);
        radar = robot.getDevice(Radar.class);
        gps = robot.getDevice(Gps.class);

        Driver driver = new Driver(robot);
        float x = gps.getPosition().getX();
        float y = gps.getPosition().getY();
        while (true) {
            driver.sleep(15000);
            driver.move((float)(x + Math.random() * 6 - 3), (float)(y + Math.random() * 6 - 3));
        }
    }
}
