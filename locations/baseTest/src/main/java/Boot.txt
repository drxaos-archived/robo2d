import com.robotech.military.api.*;

public class Boot implements Program {

    Robot robot;
    Chassis chassis;
    Radar radar;

    @Override
    public void run(Robot robot) {
        this.robot = robot;
        chassis = robot.getDevice(Chassis.class);
        radar = robot.getDevice(Radar.class);

        if (chassis == null || radar == null) {
            robot.debug("fail");
            return;
        }

        chassis.setLeftAcceleration(40d);
        chassis.setRightAcceleration(10d);
    }
}
